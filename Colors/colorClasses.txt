class White():
    def change(self):
        global FILLED_COLOR_BG
        global FILLED_COLOR_BORDER
        FILLED_COLOR_BG = "white"
        FILLED_COLOR_BORDER = "white"

    def button():
        white = Button(app, text = "White")
        white.bind("<Button-1>", White.change)
        white.pack()
